* 配置
  默认配置文件为project root下的 ~webpack.config.js~ ，可通过 ~webpack --config /path/to/config.js~ 的方式指定配置文件。

** 最基本的配置示例
   1) 单页面
      #+begin_example js
      const path = require("path")

      module.exports = {
        mode: "prodcution",
        // 指定入口文件路径
        entry: {
          path: path.resolve(__dirname, "src"),
          filename: "index.js"
        },
        output: {
          // 指定打包输出路径(path/filename)为./dist/bundle.js
          path: path.resolve(__dirname, "dist"),
          filename: "bundle.js",
          // 设置所有对打包资源URL的前缀
          // 如打包资源打包后路径为/img/main.css，那么页面引用时则变为http://www.cdn.com/img/main.css
          publicPath: "http://www.cdn.com"
        }
      }
      #+end_example
   2) 多页面
      #+begin_example js
      const path = require("path")
      const HtmlWebpackPlugin = require("html-webpack-plugin")

      module.exports = {
        mode: "prodcution",
        entry: {
          app: "./src/index.js",    // 指定入口文件路径
          common: "./src/common.js" // 指定另一个入口文件路径
        }
        output: {
          // 指定打包输出路径(path/filename)
          path: path.resolve(__dirname, "dist"),
          filename: "[name].js" // 通过占位符[name]来根据entry的键值对自动展开为多个输入路径，./dist/app.js和./dist/common.js
        },
        plugins: [
          new HtmlWebpackPlugin({
            template: "./index.html",
            filename: "app.html",
            chunks: ["app"] // 这样app.html中只引入app.js
          }),
          new HtmlWebpackPlugin({
            template: "./common.html",
            filename: "common.html",
            chunks: ["app", "common"] // 这样common.html中引入app.js和common.js
          })
        ]
      }
      #+end_example
* TODO Loaders
  Webpack内置支持JS和JSON两种文件类型，其他文件类型则要通过其他Loaders转化为有效模块，并添加到依赖图中。
  Loader本身为一个函数，输入为源文件或上一个Loader的处理结果，返回值为Loader的处理结果。
** 特点
   1) 第一个loader要返回js脚本
   2) 每个loader只做一件事，为了使loader在更多场景链式使用
   3) 每个loader都是一个模块
   4) 每个loader都是无状态的，确保loader在不同模块转换间不保存状态

** 按执行顺序分类
   1) Pre Loader
      #+begin_example js
      // 默认情况下loader的执行顺序是自底向上，从右往左。即先执行style-loader，再执行css-loader。
      // 通过enforce属性可以设置它们的类型，从而改变执行顺序
      rules: [
        {
          test: /\.css$/,
          use: "css-loader",
          enforce: "pre"
        },
        {
          test: /\.css$/,
          use: "style-loader",
          enforce: "post"
        }
      ]
      #+end_example
   2) Normal Loader
   3) Inline Loader
      #+begin_example js
      import $ from "expose-loader?$!jquery"; // expose-loader是将处理结果暴露到全局对象下(global或window)，?后的为模块元素，!后的为模块名称。
      #+end_example
      1) ~-!~ 作为前缀表示这个Inline Loader不会触发Pre和Normal Loader，仅执行Inline Loader和Post Loader；
      2) ~!~ 作为前缀表示这个Inline Loader不会触发Normal Loader，仅执行Pre Loader、Inline Loader和Post Loader；
      3) ~!!~ 作为前缀表示这个Inline Loader仅执行Inline Loader不会执行Pre、Normal 和 Post Loader。
   4) Post Loader

** Loader的组成
   1) Pitch Loader
   2) Normal Loader
   3) 多个Loader的执行流程
      #+begin_example js
      rules: [
        {
          test: /\.js$/,
          use: [
            "loader1",
            "loader2",
            "loader3"
          ]
        }
      ]
      #+end_example
      1) Pitch Loader均没有返回值
         | Pitch Loader  | pitch loader1 | -> | pitch loader2 | -> | pitch loader3 | -> | resource |
         |---------------+---------------+----+---------------+----+---------------+----+----------|
         | Normal Loader | loader1       | <- | loader2       | <- | loader3       | <- | resouce  |
      2) 某个Pitch Loader有返回值
         | Pitch Loader  | pitch loader1 | -> | pitch loader2(有返回值) |
         |---------------+---------+----+-------------------|
         | Normal Loader | loader1 | <- | pitch loader2(有返回值) |

** 常用Loader
   1) babel-loader，转换ES6/7等JS新特性
   2) css-loader，将.css文件内容转换为commonjs对象
   3) style-loader，将commonjs对象生成css内容并添加到页面的 ~<style>~ 元素中
   4) less-loader，将.less文件转换为.css文件
   5) ts-loader，将TypeScript转换为JavaScript
   6) file-loader，读取图片、字体等二进制资源
   7) url-loader，读取图片、字体等二进制资源，并提供更灵活的设置可根据图片大小生成base64字符串内嵌到页面中
   8) raw-loader，将文件以字符串的形式导入
   9) thread-loader，多进程打包JS和CSS

** 自定义Loader
   1) babel-loader
      1) 安装依赖
         #+begin_example js
         npm install @babel/core
         #+end_example
      2) 自定义Loader
         #+begin_example js
         const babel = require("@babel/core")
         const loaderUtils = require("loaderUtils")

         function loader(source){
           // this为Loader Context
           const loaderContext = this
           const options = loader.getOptions(loaderContext)
           const cb = loaderContext.async()
           babel.transform(source,
                           {
                            ...options,
                            sourceMap: true, // 让babel生成source map，但要在webpack.config.js中设置devtool才能输出source map
                            filename: loaderContext.resourcePath.split("/").pop() // 设置source map文件的名称
                           },
                           function(err, result){
                             cb(err, result.code, result.map)
                           })
         }

         module.exports = loader
         #+end_example

   2) banner-loader
      1) 自定义Loader
         #+begin_example js
         const fs = require("fs")
         const loaderUtils = require("loaderUtils")
         const validateOptions = require("schema-utils")

         const schema = {
           type: "object",
           properties: {
             text: {
               type: "string"
             },
             filename: {
               type: "string"
             }
           }
         }

         function loader(source){
           this.cacheable && this.cacheable() // 若存在缓存则直接返回，缓存默认开启
           // this.cacheable(false) 关闭缓存
           const options = loaderUtils.getOptions(this)
           validateOptions(schema, options, "banner-loader")
           const cb = this.async()
           if (options.filename) {
             this.addDependency(options.filename) // 将文件添加到webpack的依赖中，当该文件发生变化时会触发重新打包
             fs.readFile(options.filename, "utf8", function(err, data){
               cb(err, `/**${data}**/${source}`)
             })
           }
           else{
             cb(null, `/**${options.text}**/${source}`)
           }
         }

         module.exports = loader
         #+end_example
   3) file-loader
      将文件复制到output目录下，并返回一个md5的文件名。
      #+begin_example js
      const loaderUtils = require("loader-utils")

      function loader(binary){
        const filename = loaderUtils.interpolateName(this,
                                                     "[hash].[ext]",
                                                     {content: binary})
        this.emitFile(filename, binary)

        return `module.exports = "${filename}"`
      }
      loader.raw = true // 以二进制形式读取文件内容

      module.exports = loader
      #+end_example
   4) url-loader
      #+begin_example js
      const loaderUtils = require("loader-utils")
      const mime = require("mime")

      function loader(binary){
        const {limit} = loaderUtils.getOptions(this)
        if (limit && limit > binary.length) {
          return `module.exports = "data:${mime.getType(this.resoucePath)};base64,${binary.toString("base64")}"`
        }
        else{
          return require("./file-loader").call(this, binary)
        }
       }
      loader.raw = true // 以二进制形式读取文件内容

      module.exports = loader
      #+end_example
   5) less-loader,css-loader,style-loader
      #+begin_example js
      // ./less-loader.js
      const less = require("less")

      function loader(source){
        let css
        less.render(source, function(err, result){
          css = result.css
        })

        return css
      }

      module.exports = loader
      #+end_example
      #+begin_example js
      // ./css-loader.js
      const reg = /url\((.+?)\(/g

      function loader(source){
        let pos = 0
        let curr = null
        let segs = []
        while (curr = reg.exec(source)){
          let [matchUrl, g] = curr
          let last = reg.lastIndex - matchUrl.length
          segs.push(source.slice(pos, last))
          pos = reg.lastIndex
          segs.push(`"url("+require(${g})+")"`)
        }

        segs.push(JSON.stringify(source.slice(pos)))

        return `module.exports=${segs.join("")}`
      }

      module.exports = loader
      #+end_example
      #+begin_example js
      // ./style-loader.js

      function loader(source){
        let str = `
          var style = document.createElement("style")
          style.innerHTML = ${JSON.stringify(source)}
          document.head.appendChild(style)
        `

        return str
      }

      const loaderUtils = require("loader-utils")
      loader.pitch = function(remainingRequest){
        // remainingRequest为剩余的loader，值为css-loader!less-loader!./style.less
        let str = `
          var style = document.createElement("style")
          style.innerHTML = ${require(loaderUtils.stringifyRequest(this, "!!" + remainingRequest))}
          document.head.appendChild(style)
        `

        return str
      }

      module.exports = loader
      #+end_example

* Plugins
   用于bundle文件的优化、资源管理和环境变量注入等，作用于整个构建过程。可以说Loader实现不到的功能都是通过Plugin来实现的。

** 常用的Plugin
   1) CommonsChunkPlugin，将chunks相同的模块代码提取成公共js
   2) CleanWebpackPlugin，清理构建目录
   3) ExtractTextWebpackPlugin，将CSS从bundle文件里提取成一个独立的CSS文件
   4) CopyWebpackPlugin，将文件或目录拷贝到构建目录中
   5) HtmlWebpackPlugin，创建hmtl文件来承载输出的bundle
   6) UglifyjsWebpackPlugin，压缩JS
   7) ZipWebpackPlugin，将打包出的资源生成一个zip
   8) bundle-buddy-webpack-plugin，用于发现多个JavaScript Chunk/Splits中重复冗余源代码

** 配置使用
   1) 基本配置方式
      #+begin_example js
      const HtmlWebpackPlugin = require("html-webpack-plugin")

      module.exports = {
        entry: "./src/index.js",
        output: "./dist/bundle.js",
        plugins: [

            filename: "index.html", // 打包后的文件名
            hash: true, // 引入的bundle.js中加入hash值，每次打包都不同。
            // 压缩html

              collapseWhitespace: true, // 变成一行
            }

   2) 热更新WDS(Webpack-dev-server)
      - 特点：
        1) 不刷新浏览器
        2) 不输出文件，而是放在内存中加速构建速度
      - 依赖：
        1) webpack-dev-server
        2) HotModuleReplacementPlugin(Webpack内置)
      - 使用
        - 安装依赖
          #+begin_example bash
          npm install webpack-dev-server --save-dev
          #+end_example
        - 配置
          1) package.json
             #+begin_example json
             "scripts": {
               "dev": "webpack-dev-server --open" // 自动启动浏览器
             }
             #+end_example
          2) webpack.config.js
             #+begin_example js
             const webpack = require("webpack")

             module.exports = {
               mode: "development",
               entry: "./src/index.js",
               output: "./dist/bundle.js",
               plugins: [
                 new webpack.NamedModulesPlugin(), // 打印更新的模块路径
                 new webpack.HotModuleReplacementPlugin() // 热更新
               ],
               // 配置webpack dev server
               devServer: {
                 port: 3000, // 设置端口
                 progress: true, // 显示进度条
                 compress: true, // 启用GZIP压缩
                 contentBase: "./dist", // 设置网站内容的根目录
                 hot: true, // 是否开启热更新
                 // 设置日志输出级别
                 // errors-only 只在发生错误时输出
                 // minimal 只在发生错误或有新的编译时输出
                 // none 没有输出
                 // normal 标准输出
                 // verbos 全部输出
                 stats: "errors-only",
                 // 网络代理，处理跨域问题
                 proxy: {
                   "/api": "http://localhost:3000" // 当请求以/api开头则代理到http://localhost:3000/api开头的服务
                   // 当请求以/apis开头则代理到http://localhost:3000/user开头的服务
                   "/apis": {
                     target: "http://localhost:3000",
                     pathRewrite: {
                       "/apis": "/user"
                     }
                   }
                 },
                 // 数据模拟
                 before(app){
                   app.get("/user", (req, res) => {
                     res.json({name: "test"})
                   })
                 }
               }
             }
             #+end_example
      - 原理：
        服务端加入HMR Server，而客户端加入HMR Runtime。HMR Server负责将热更新的文件输出给HMR Runtime，而HMR Runtime则负责更新变化的部分而修改其它代码。

* 文件指纹

** 指纹类型
   1) hash
      和整个项目关联的hash，项目下只要有文件发生变化，那么整个项目的hash值将会发生变化。
   2) chunk hash
      和webpack打包的chunk相关，不同的entry会生成不同的chunk hash。若该entry下的文件发生变化，则该chunk的hash值将发生变化，但其他的entry的hash值不变。
   3) content hash
      一般CSS样式会从代码中分离出来加载，因此会对样式文件采用content hash，根据该文件的内容变化生成hash。

** 使用示例
   #+begin_example js
   const MiniCssExtractPlugin = require("mini-css-extract-plugin")

   module.exports = {
     entry: "./src/index.js",
     output: {
       path: __dirname + "/dist",
       filename: "[name][chunkhash:8].js" // 占位符chunkhash:8表示生成chunkhash值并截取前8个字符。
     },
     module: {
       rules: [
         {
           test: /\.css$/,
           use: [
             MiniCssExtractPlugin.loader,
             "style-loader",
             "css-loader"
           ]
         },
         { // 图片、字体等文件的生成content hash
           // file-loader的占位符
           // [ext] - 资源后缀名
           // [name] - 资源文件名
           // [path] - 资源的相对路径
           // [folder] - 资源所在的目录名称
           // [contenthash] 或 [hash] - content hash
           // [emoji] - 一个随机指代文件内容的emoji
           test: /\.(png|svg|jpg|git)$/,
           use: [{
             loader: "file-loader",
             options: {
               name: "img/[name][hash:8].[ext]"
             }
           }]
         }
       ]
     },
     plugins: [
       // 该插件用于将style元素中的样式提取为样式文件，然后通过link元素加入到页面上。
       new MiniCssExtractPlugin({
         filename: "[name][contenthash:8].css",
         chunkFilename: "[id].css" // TODO
       })
     ]
   }
   #+end_example

          filename: "_dll_[name].js",
          library: "_dll[name]"

          new webpack.DllPlugin({
            name: "_dll[name]", // 和library值一样

       #+begin_quote
       What this command does is to increase the number of watches allowed for a single user. By the default the number can be low (8192 for example). When nodemon tries to watch large numbers of directories for changes it has to create several watches, which can surpass that limit.
       #+end_quote
