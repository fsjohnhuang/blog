** nvm
   官网：https://github.com/nvm-sh/nvm

   node.js多版本管理工具：安装/卸载node.js，切换node.js版本。

   1) 安装nvm
      #+begin_example bash
      export NVM_DIR="$HOME/.nvm" && (
        git clone https://github.com/nvm-sh/nvm.git "$NVM_DIR"
        cd "$NVM_DIR"
        git checkout `git describe --abbrev=0 --tags --match "v[0-9]*" $(git rev-list --tags --max-count=1)`
      ) && \. "$NVM_DIR/nvm.sh"
      #+end_example
   2) 配置开机后启用nvm
      在~/.bashrc,~/.profile或~/.zshrc中加入如下内容
      #+begin_example bash
      export NVM_DIR="$HOME/.nvm"
      [ -s "$NVM_DIR/nvm.sh" ] && \. "$NVM_DIR/nvm.sh" # This loads nvm
      #+end_example
   3) 更新nvm
      #+begin_example bash
      (
        cd "$NVM_DIR"
        git fetch --tags origin
        git checkout `git describe --abbrev=0 --tags --match "v[0-9]*" $(git rev-list --tags --max-count=1)`
      ) && \. "$NVM_DIR/nvm.sh"
      #+end_example
   4) 使用
      1) 安装node.js
         #+begin_example bash
         # use a mirror of the node binaries
         export NVM_NODEJS_ORG_MIRROR=https://nodejs.org/dist

         nvm install node # the latest version
         nvm install 6.14.4 # a specific version
         #+end_example
      2) 设置默认版本
         #+begin_example bash
         nvm alias default 8.16.0
         #+end_example
      3) 切换版本
         #+begin_example bash
         nvm use 7.10.0
         #+end_example
      4) 不切换版本，但在指定版本下运行命令
         #+begin_example bash
         nvm exec 4.2 node --version
         #+end_example
      5) 查看可下载的版本
         #+begin_example bash
         nvm ls-remote
         #+end_example
      6) 获取指定版本的可执行文件的安装路径
         #+begin_example bash
         nvm which 5.0
         #+end_example
      7) 恢复环境变量PATH
         #+begin_example bash
         nvm deactivate
         #+end_example
   5) 版本代号
      1) ~node~ 表示最新版本的node
      2) ~iojs~ 表示最新版本的io.js
   6) ~.nvmrc~
      place in the project root directory to specify the node version to use under this directory.
      - 创建方式
        #+begin_example bash
        echo "5.9" > .nvmrc
        echo "node" > .nvmrc
        #+end_example
** nrm
NPM registry 管理工具，能够查看和切换当前使用的registry。
1) 安装
   #+begin_example bash
   npm install nrm -g
   #+end_example
2) 使用
   #+begin_example bash
   # 查看可用的registry
   nrm ls

   # 切换registry
   nrm use cnpm

   # 访问registry的主页
   nrm home

   # 添加源
   nrm add <registry_name> <registry_url>
   #+end_example

** 网络代理配置
 1) 全局配置
    #+begin_example bash
    # 设置http和https的网络代理
    npm config set proxy=http://[username:password@]<proxy_ip>

    # 设置http的网络代理
    npm config set http-proxy=http://[username:password@]<proxy_ip>

    # 设置https的网络代理
    npm config set https-proxy=http:///[username:password@]<proxy_ip>

    # 取消代理
    npm config delete proxy
    npm config delete http-proxy
    npm config delete https-proxy

    # 获取设置的值
    npm config get proxy
    npm config get http-proxy
    npm config get https-proxy
    #+end_example
 2) 当前操作配置网络代理
    #+begin_example bash
    npm --proxy=http://<proxy_ip> <command>
    npm --http-proxy=http://<proxy_ip> <command>
    npm --https-proxy=http://<proxy_ip> <command>
    #+end_example

** precommit钩子
1) 通过lint-staged增量检查修改的文件
   1) 安装依赖
      #+begin_example bash
      npm install husky --save-dev
      #+end_example
   2) 配置package.json
      #+begin_example
      // package.json
      "scripts": {
        "precommit": "lint-staged"
      },
      "lint-staged": {
        "linters": {
          "*.{js,scss}": ["eslint --fix", "git add"]
        }
      }
      #+end_example
